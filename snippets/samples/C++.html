set<Tuple> Relation::selectConst(set<Tuple> results, int position, string value) {
    for (set<Tuple>::iterator it = results.begin(); it != results.end(); ) {
        Tuple t = *it;
        if (t.at(position) != value) results.erase(it++);
        else ++it;
	}
    return results;
}

set<Tuple> Relation::selectVar(set<Tuple> results, int position1, int position2) {
    for (set<Tuple>::iterator it = results.begin(); it != results.end(); ) {
        Tuple t = *it;
        if (t.at(position1) != t.at(position2)) results.erase(it++);
        else ++it;
	}
    return results;
}

set<Tuple> Relation::project(set<Tuple> results, vector<int> positions) {
    set<Tuple> newResults;
    for (set<Tuple>::iterator it = results.begin(); it != results.end(); ++it) {
        Tuple t = *it;
        newResults.insert(t.project(positions));
	}
    return newResults;
}

void Relation::rename(int position, string value) {
    schema.at(position) = value;
}
